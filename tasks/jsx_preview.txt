  const [showBridge, setShowBridge] = useState(false);
  const [showTagline, setShowTagline] = useState(false);
  const [testimonials, setTestimonials] = useState([]);
  const [jobs, setJobs] = useState([]);
  const [schools, setSchools] = useState(FALLBACK_SCHOOLS);
  const [academies, setAcademies] = useState(FALLBACK_ACADEMIES);
  const [heroStyle, setHeroStyle] = useState({ opacity: 1, transform: 'scale(1)' });
  const heroRef = useRef(null);
  const sectionRefs = useRef([]);

  // Bridge + tagline 등장
  useEffect(() => {
    const t1 = setTimeout(() => setShowBridge(true), 50);
    const t2 = setTimeout(() => setShowTagline(true), 50);
    return () => { clearTimeout(t1); clearTimeout(t2); };
  }, []);

  // 실제 API — testimonials
  useEffect(() => {
    fetch(`${BASE}/api/testimonials?limit=6&random=1`)
      .then(r => r.json())
      .then(d => {
        if (d?.success && d.data?.testimonials?.length > 0) {
          setTestimonials(d.data.testimonials.map(t => ({
            text: t.review_text, stars: t.rating, name: t.name, country: t.country,
          })));
        } else { setTestimonials(FALLBACK_TESTIMONIALS); }
      })
      .catch(() => setTestimonials(FALLBACK_TESTIMONIALS));
  }, []);

  // 실제 API — jobs
  useEffect(() => {
    fetch(`${BASE}/api/jobs?limit=20`)
      .then(r => r.json())
      .then(d => {
        const all = d?.data ?? [];
        setJobs(all.length > 0 ? all.slice(0, 4) : FALLBACK_JOBS);
      })
      .catch(() => setJobs(FALLBACK_JOBS));
  }, []);

  // 실제 API — partners
  useEffect(() => {
    fetch(`${BASE}/api/partners`)
      .then(r => r.json())
      .then(d => {
        const partners = d?.data?.partners ?? [];
        if (partners.length > 0) {
          const acs = partners.filter(p => p.category === 'academy').map(p => p.name);
          const scs = partners.filter(p => p.category === 'school').map(p => p.name);
          if (acs.length > 0) setAcademies(acs);
          if (scs.length > 0) setSchools(scs);
        }
      })
      .catch(() => {});
  }, []);

  // Hero parallax (scroll → opacity + scale)
  useEffect(() => {
    const onScroll = () => {
      if (!heroRef.current) return;
      const rect = heroRef.current.getBoundingClientRect();
      const h = heroRef.current.offsetHeight;
      const progress = Math.max(0, Math.min(1, -rect.top / (h * 0.6)));
      setHeroStyle({
        opacity: 1 - progress,
        transform: `scale(${1 - progress * 0.05})`,
      });
    };
    window.addEventListener('scroll', onScroll, { passive: true });
    return () => window.removeEventListener('scroll', onScroll);
  }, []);

  // 섹션 fade-in (IntersectionObserver)
  useEffect(() => {
    const obs = new IntersectionObserver(
      entries => entries.forEach(e => {
        if (e.isIntersecting) e.target.classList.add('visible');
      }),
      { threshold: 0.15, rootMargin: '0px 0px -40px 0px' }
    );
    sectionRefs.current.forEach(el => el && obs.observe(el));
    return () => obs.disconnect();
  }, [testimonials, jobs]);

  const addRef = el => { if (el && !sectionRefs.current.includes(el)) sectionRefs.current.push(el); };
  const formatAge = ages => ages.map(a => AGE_LABELS[a] || a).join(' · ');

  return (
    <div style={{ fontFamily:"'Pretendard Variable',Pretendard,-apple-system,sans-serif", overflowX:'hidden' }}>
      <style>{css}</style>

      {/* ── NAVBAR ── */}

          <span style={{ fontSize:21, fontWeight:600, letterSpacing:0, color:'#1d1d1f', marginRight:'auto', cursor:'pointer' }}>BRIDGE</span>

            {NAV_LINKS.map(l => <a key={l} href="#" className="nav-link">{l}</a>)}

          </div>
        </div>
      </nav>

                <stop offset="0.5" stopColor="white" stopOpacity="0.35"/>

                <animate attributeName="x2" values="0;2100" dur="3.5s" repeatCount="indefinite" begin="3s"/>
              </linearGradient>
            </defs>
            <g className="bridge-arc-group">

            </g>
            <path d="M 0 520 Q 700 180, 1400 520" className="bridge-sweep" stroke="rgba(255,255,255,0.5)" strokeWidth={3} strokeLinecap="round" filter="url(#sweepGlow)"/>
            <line x1={420} y1={520} x2={420} y2={280} className="bridge-tower" stroke="white" strokeWidth={2} strokeLinecap="round"/>
            <line x1={980} y1={520} x2={980} y2={280} className="bridge-tower" stroke="white" strokeWidth={2} strokeLinecap="round"/>
            <line x1={420} y1={280} x2={280} y2={410} className="bridge-stay bridge-stay-l1" stroke="white" strokeWidth={0.8} strokeLinecap="round" style={{'--len':'191'}}/>
            <line x1={420} y1={280} x2={200} y2={465} className="bridge-stay bridge-stay-l2" stroke="white" strokeWidth={0.8} strokeLinecap="round" style={{'--len':'287'}}/>
            <line x1={980} y1={280} x2={1120} y2={410} className="bridge-stay bridge-stay-r1" stroke="white" strokeWidth={0.8} strokeLinecap="round" style={{'--len':'191'}}/>
            <line x1={980} y1={280} x2={1200} y2={465} className="bridge-stay bridge-stay-r2" stroke="white" strokeWidth={0.8} strokeLinecap="round" style={{'--len':'287'}}/>
            <path d="M 0 520 Q 700 180, 1400 520" stroke="url(#postSweep)" strokeWidth={4} strokeLinecap="round" opacity={0.5} filter="url(#cableGlow)"/>

        </div>

        {/* BRIDGE 텍스트 — parallax */}
        <div style={{ position:'relative', zIndex:10, textAlign:'center', padding:'0 16px', ...heroStyle }}>

            BRIDGE

          <p style={{ fontSize:'clamp(18px,2.5vw,28px)', fontWeight:300, letterSpacing:'-0.01em' }}>
            {'A career that changes your life'.split('').map((c,i) => (

                style={{ opacity:showTagline?undefined:0, animationDelay:showTagline?`${i*0.025}s`:undefined, display:'inline-block', minWidth:c===' '?'0.3em':undefined }}>

        </div>

        {/* 스크롤 인디케이터 */}
        <div style={{ position:'absolute', bottom:32, left:'50%', transform:'translateX(-50%)', display:'flex', flexDirection:'column', alignItems:'center', gap:12, zIndex:10 }}>
          <span className="scroll-pulse" style={{ fontSize:12, letterSpacing:'0.3em', textTransform:'uppercase', color:'#636366', fontWeight:600 }}>Scroll</span>
          <div className="scroll-bounce">
            <svg width="24" height="24" fill="none" viewBox="0 0 24 24" stroke="#636366">

        <section style={{ background:'#000', padding:'96px 16px', borderTop:'1px solid rgba(255,255,255,0.06)' }}>
          <div style={{ maxWidth:1120, margin:'0 auto' }}>
            <div ref={addRef} className="fade-section" style={{ textAlign:'center', marginBottom:64 }}>
              <p style={{ color:'#2997ff', fontSize:13, fontWeight:600, textTransform:'uppercase', letterSpacing:'0.2em', marginBottom:12 }}>Testimonials</p>
              <h2 style={{ fontSize:'clamp(28px,4vw,42px)', fontWeight:700, color:'white', letterSpacing:'-0.02em' }}>Hear from Our Teachers</h2>
            </div>
            <div ref={addRef} className="fade-section" style={{ display:'grid', gridTemplateColumns:'repeat(3,1fr)', gap:20 }}>
              {testimonials.slice(0,6).map((t,i) => (
                <div key={i} className="testimonial-glass">

                      {[0,1,2,3,4].map(s => (
